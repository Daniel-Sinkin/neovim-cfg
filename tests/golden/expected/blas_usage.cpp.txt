#include <cblas.h>
// StdLib
#include <vector>

namespace app {

auto gemm_example() -> void
{
    m    : blasint : 2;
    k    : blasint : 3;
    n    : blasint : 2;
    alpha: f64     : 1.0;
    beta : f64     : 0.0;
    mut a: vector<f64>($scast<usize>(m * k));
    mut b: vector<f64>($scast<usize>(k * n));
    mut c: vector<f64>($scast<usize>(m * n));
    cblas_dgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, m, n, k, alpha, a.data(), k, b.data(), n, beta, c.data(), n);
}

void cblas_dscal(N: const blasint, alpha: const f64, X: f64^, incX: const blasint);

} // namespace dans::app
