auto Camera::configure(const CameraConfig& config) $ne -> mut Camera&
{
    pivot_ = config.pivot;
    distance_ = config.distance;
    yaw_ = config.yaw;
    pitch_ = config.pitch;
    fov_y_ = config.fov_y;
    orbit_sensitivity_ = config.orbit_sensitivity;
    pivot_sensitivity_ = config.pivot_sensitivity;
    zoom_sensitivity_ = config.zoom_sensitivity;
    z_near_ = config.z_near;
    z_far_ = config.z_far;
    allow_pivot_move_ = config.allow_pivot_move;
    clamp_position_z_min_ = config.clamp_position_z_min;
    min_position_z_ = config.min_position_z;
    projection_mode_ = config.projection_mode;
    apply_constraints();
    return *this;
}

auto Camera::set_pivot(const Vec3 pivot) $ne -> mut Camera&
{
    pivot_ = pivot;
    apply_constraints();
    return *this;
}

auto Camera::translate_pivot(const Vec3 offset) $ne -> mut Camera&
{
    return set_pivot(pivot_ + offset);
}

auto Camera::apply_constraints() $ne -> void
{
    if (!clamp_position_z_min_)
    {
        return;
    }
    min_offset_z := min_position_z_ - pivot_.z;
    if (min_offset_z <= -distance_)
    {
        return;
    }
    auto min_pitch =
        asin(clamp(min_offset_z / max(distance_, 0.001f), -1.0f, 1.0f));
    pitch_ = max(pitch_, min_pitch);
}

auto Camera::position() const $ne -> Vec3
{
    cos_pitch := cos(pitch_);
    Vec3 offset{
        distance_ * cos_pitch * cos(yaw_),
        distance_ * cos_pitch * sin(yaw_),
        distance_ * sin(pitch_),
    };
    return pivot_ + offset;
}

auto Camera::view_matrix() const $ne -> Mat4
{
    return glm::lookAt(position(), pivot_, k_axis_z);
}

auto Camera::projection_matrix(f32 aspect) const $ne -> Mat4
{
    clamped_aspect := max(0.01f, aspect);
    switch (projection_mode_)
    {
        case ProjectionMode::orthographic:
            {
                half_height := 0.5f * view_height();
                half_width := half_height * clamped_aspect;
                auto proj = glm::orthoRH_ZO(
                    -half_width, half_width, -half_height, half_height, z_near_, z_far_
                );
                proj[1][1] *= -1.0f;
                return proj;
            }
        case ProjectionMode::perspective:
            {
                mut proj := glm::perspective(fov_y_, clamped_aspect, z_near_, z_far_);
                proj[1][1] *= -1.0f;
                return proj;
            }
    }
    return Mat4{1.0f};
}

auto Camera::view_projection_matrix(f32 aspect) const $ne -> Mat4
{
    return projection_matrix(aspect) * view_matrix();
}

auto Camera::right() const $ne -> Vec3
{
    forward := normalize_or(pivot_ - position(), -k_axis_y);
    return normalize_or(glm::cross(forward, k_axis_z), k_axis_x);
}

auto Camera::up() const $ne -> Vec3
{
    forward := normalize_or(pivot_ - position(), -k_axis_y);
    return normalize_or(glm::cross(right(), forward), k_axis_z);
}

auto Camera::view_height() const $ne -> f32
{
    return 2.0f * distance_ * tan(0.5f * fov_y_);
}

auto Camera::units_per_pixel_y(f32 viewport_height_px) const $ne -> f32
{
    return view_height() / max(1.0f, viewport_height_px);
}

auto Camera::pan_offset_world(f32 dx_px, f32 dy_px, f32 viewport_height_px) const $ne -> Vec3
{
    units := units_per_pixel_y(viewport_height_px);
    return (-dx_px * units) * right() + (dy_px * units) * up();
}

auto make_camera_ray(const Camera& camera, Vec2 cursor_logical_px, Vec2 viewport_logical_px) $ne
    -> Ray
{
    viewport := glm::max(viewport_logical_px, Vec2{1.0f});
    ndc_x := 2.0f * cursor_logical_px.x / viewport.x - 1.0f;
    ndc_y := 2.0f * cursor_logical_px.y / viewport.y - 1.0f;
    aspect := viewport.x / viewport.y;
    inverse_view_projection := glm::inverse(camera.view_projection_matrix(aspect));
    near_clip: Vec4 = ndc_x, ndc_y, 0.0f, 1.0f;
    far_clip : Vec4 = ndc_x, ndc_y, 1.0f, 1.0f;
    mut near_world := inverse_view_projection * near_clip;
    mut far_world := inverse_view_projection * far_clip;
    near_world /= near_world.w;
    far_world /= far_world.w;
    origin: Vec3 = near_world;
    return Ray{
        origin,
        direction = normalize_or(Vec3{far_world - near_world}, -k_axis_z),
    };
}
